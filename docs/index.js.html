  cors({
    origin: [
      "http://localhost:5000",
      "https://auth-todo-list-eight.vercel.app",
      "https://auth-todo-list-vitaliykalachev.vercel.app/",
    ],
    credentials: true,
  })

app.use("/auth", authRouter);

    await mongoose.connect(

  }
};

start();
